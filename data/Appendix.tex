	\chapter{常见矢量公式}
	矢量混合积
	\[
	\vec{A}\cdot \left( \vec{B}\times \vec{C} \right) =\vec{B}\cdot \left( \vec{C}\times \vec{A} \right) =\vec{C}
	\]
	
	
	矢量二重叉积
	\[
	\vec{A}\times \left( \vec{B}\times \vec{C} \right) =\vec{B}\left( \vec{A}\cdot \vec{C} \right) -\vec{C}\left( \vec{A}\cdot \vec{B} \right) 
	\]
